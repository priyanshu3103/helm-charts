Create an Error Middleware for Traefik to handle error pages for ArgoCD. This middleware will catch specific HTTP status codes (500, 502, 503, 504) and redirect users to a custom error page served by the `error-page-svc` service on port 80.
kubectl apply -f - <<EOF
apiVersion: traefik.io/v1alpha1
kind: Middleware
metadata:
  name: error-pages
  namespace: argocd
spec:
  errors:
    status:
      - "500"
      - "502"
      - "503"
      - "504"
      - "500-599"     # catches all 5xx errors
    query: "/?code={status}"
    service:
      name: error-page-svc
      port: 80
EOF

Create a simple error page service (you have 2 options)
kubectl apply -f - <<EOF
apiVersion: v1
kind: ConfigMap
metadata:
  name: error-pages
  namespace: argocd
data:
  index.html: |
    <!DOCTYPE html>
    <html>
    <head><title>Service Error</title></head>
    <body style="font-family:Arial;text-align:center;padding-top:50px;">
      <h1>Argo CD is temporarily unavailable</h1>
      <p>Error Code: <strong>{status}</strong></p>
      <p>Please wait a moment and try again.</p>
    </body>
    </html>
---
apiVersion: apps/v1
kind: Deployment
metadata:
  name: error-page
  namespace: argocd
spec:
  replicas: 1
  selector:
    matchLabels:
      app: error-page
  template:
    metadata:
      labels:
        app: error-page
    spec:
      containers:
      - name: nginx
        image: nginx:alpine
        ports:
        - containerPort: 80
        volumeMounts:
        - name: pages
          mountPath: /usr/share/nginx/html
      volumes:
      - name: pages
        configMap:
          name: error-pages
---
apiVersion: v1
kind: Service
metadata:
  name: error-page-svc
  namespace: argocd
spec:
  selector:
    app: error-page
  ports:
  - port: 80
    targetPort: 80
EOF

Attach the middleware to your Argo CD IngressRoute (Best method)

kubectl apply -f - <<EOF
apiVersion: traefik.io/v1alpha1
kind: IngressRoute
metadata:
  name: argocd
  namespace: argocd
spec:
  entryPoints:
    - web
  routes:
  - match: Host(\`argohelm.localhost\`)
    kind: Rule
    services:
    - name: argocd-server
      port: 80
    middlewares:
    - name: error-pages
EOF



✅ Got it! You want custom error pages (502, 503, 504, etc.) for Jenkins, not Argo CD.
Here’s the complete setup for Jenkins:
1. Create Error Pages Middleware for Jenkins


kubectl apply -f - <<EOF
apiVersion: traefik.io/v1alpha1
kind: Middleware
metadata:
  name: error-pages
  namespace: jenkins          # ← Change if your Jenkins namespace is different
spec:
  errors:
    status:
      - "500"
      - "502"
      - "503"
      - "504"
      - "500-599"
    query: "/?code={status}"
    service:
      name: error-page-svc
      port: 80
EOF

kubectl apply -f - <<EOF
apiVersion: v1

    </html>

apiVersion: apps/v1

  replicas: 1

3. Apply Middleware to Jenkins Ingress / IngressRoute
Option A: Using IngressRoute (Recommended for Traefik)

kubectl apply -f - <<EOF

metadata:
  name: jenkins
  namespace: jenkins
spec:
  entryPoints:
    - web
